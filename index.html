Vui lòng thử lại!'); }; } function processCommand(command) { let response = ''; // Simple command processing if (command.includes('xin chào') || command.includes('hello')) { response = 'Xin chào! Tôi là Yohu Smart, tôi có thể giúp gì cho bạn?'; } else if (command.includes('thời tiết')) { response = 'Thời tiết hôm nay tại TP.
